<html>
<head>
<title>Digital Clock</title>

</head>
<body onload = "startTime()">

<div id="theTime"></div>

<script type = "text/javascript">
function startTime(){
var d
d = new Date()
h = d.getHours()
m = d.getMinutes()
s = d.getSeconds()
document.getElementById("theTime").innerHTML = h + ":" + m + ":" + s
setInterval("showTime()",1000)
}

function showTime(){
var d
d = new Date()
h = d.getHours()
m = d.getMinutes()
s = d.getSeconds()
document.getElementById("theTime").innerHTML = h + ":" + m + ":" + s
}

</script>
</body>

function validate(){

var n , e, p
n = document.dataform.uname.value
e = document.dataform.uemail.value
p = document.dataform.upwd.value

//name is mandatory field

if(n.length == 0){

alert("please enter name")
document.dataform.uname.focus()
return
}
//email is mandatory
if(e.length == 0){

alert("please enter email")
document.dataform.uemail.focus()
return
}
//password is mandatory
if(p.length == 0){

alert("please enter password")
document.dataform.upwd.focus()
return
}

//email validation :
//valid chars abcdefghijklmnopqrstuvqxyz.@_0123456789
//@ should appear once and only once
var validChars = "abcdefghijklmnopqrstuvqxyz.@_0123456789"
for(i = 0 ; i < e.length ; ++i){
if(validChars.indexOf(e.charAt(i)) == -1 ){
alert("enter valid email characters")
document.dataform.uemail.focus()
return
}
}
if((e.indexOf("@") != e.lastIndexOf("@")) || e.indexOf("@") == -1){
alert("enter valid email address: @ sign should appear once and only once")
document.dataform.uemail.focus()
return
}

//password should be more than six characters

if(p.length < 6 ){
alert("the password should be minimum six characters")
document.dataform.upwd.focus()
return
}
//if everything is validated, submit the form
document.dataform.submit()
}

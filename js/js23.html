function result(){
var n1,n2,op
n1 = parseInt(document.dataform.num1.value)
n2 = parseInt(document.dataform.num2.value)
op = document.dataform.theOp.selectedIndex

switch(op){
case 0:
//plus
document.dataform.theResult.value = n1 + n2
break
case 1:
//minus
document.dataform.theResult.value = n1 - n2
break
case 2:
//multiply
document.dataform.theResult.value = n1 * n2
break

case 3:
//divide
document.dataform.theResult.value = n1 / n2
break

case 4:
//modulus
document.dataform.theResult.value = n1 % n2
break
}
}
